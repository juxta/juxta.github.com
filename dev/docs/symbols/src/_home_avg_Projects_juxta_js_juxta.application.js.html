1 Juxta.Application = $.Class({
  2 	settings: {
  3 		closable: false,
  4 		maximized: false
  5 	},
  6 	init: function(element, options) {
  7 		this.settings = $.extend({}, this.settings, options);
  8 		
  9 		this.$application = $(element);
 10 		this.$menu = this.$application.find('.menu');
 11 		this.$statusBar = this.$application.find('.status');
 12 		
 13 		this.tune(this.settings);
 14 		
 15 		if (this.settings.closable) {
 16 			this.$application.find('.close').show();
 17 			this.$application.find('.close').click(function() { history.back(); });
 18 		} else{
 19 			this.$application.find('.close').hide();
 20 		}
 21 	},
 22 	tune: function(options) {
 23 		if ($.isPlainObject(options.header)) {
 24 			this.$application.find('h1').html(
 25 				options.header.title + 
 26 				(options.header.name ? ' <a>' + options.header.name + '</a>' : '') +
 27 				(options.header.from ? ' <span class="from">from <a>' + options.header.from + '</a></span>' : '')
 28 			);
 29 		} else{
 30 			this.$application.find('h1').html(options.header);
 31 		}
 32 		this.menu(options.menu);
 33 	},
 34 	show: function(options) {
 35 		if (!options) {
 36 			Juxta.show();
 37 		} else {
 38 			options = $.extend({}, this.settings, options);
 39 			this.tune(options);
 40 		}
 41 
 42 		if (!this.$application.is(':visible')) {
 43 			$('#applications .application').hide();
 44 			this.$application.show();
 45 		}
 46 
 47 		if (this.settings.maximized) {
 48 			this.maximize();
 49 		} else{
 50 			this.restore();
 51 		}
 52 
 53 		return this;
 54 	},
 55 	hide: function() {
 56 		this.$application.hide();
 57 		return this;
 58 	},
 59 	menu: function(menu) {
 60 		this.$menu.empty();
 61 		var _this = this;
 62 		if ($.isPlainObject(menu)) {
 63 			jQuery.each(menu, function(title, action) {
 64 				if ($.isPlainObject(action)) {
 65 					_this.$menu.append('<a href="' + (action.href ? action.href : '') + (action.click ? '" onclick="' + action.click + '"' : '"') + '>' + title + '</a>');
 66 				} else if (action) {
 67 					_this.$menu.append('<a href="' + action + '">' + title + '</a>');
 68 				} else{
 69 					_this.$menu.append('<a>' + title + '</a>');
 70 				}
 71 			});
 72 		}
 73 		return this;
 74 	},
 75 	maximize: function() {
 76 		$('#sidebar').addClass('minimized');
 77 		$('#applications').addClass('maximized');
 78 		return this;
 79 	},
 80 	restore: function() {
 81 		$('#applications').removeClass('maximized');
 82 		$('#sidebar').removeClass('minimized');
 83 		return this;
 84 	},
 85 	status: function(text) {
 86 		this.$statusBar.text(text);
 87 	}
 88 });
