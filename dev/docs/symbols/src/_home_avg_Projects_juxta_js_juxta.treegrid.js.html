1 /**
  2  * @class Tree grid
  3  * @extends Juxta.Grid
  4  */
  5 Juxta.TreeGrid = function(grid) {
  6 	Juxta.Grid.call(this, grid);
  7 }
  8 
  9 Juxta.TreeGrid.prototype.init = function(grid) {
 10 
 11 	this._selectRow = Juxta.Grid.prototype.selectRow;
 12 	this._deselectRow = Juxta.Grid.prototype.deselectRow;
 13 
 14 	Juxta.Grid.prototype.init.call(this, grid);
 15 
 16 	var that = this;
 17 
 18 	this.$body.find('td.expand, td.collapse').live('click', function(event) {
 19 		// Temporary
 20 		var $target = $(event.target),
 21 			$row = $target.parents('tr');
 22 		if (!$target.parents('tr').next('tr.content').get(0)) {
 23 			$target.parents('tr').after('<tr class="content"><td colspan="99"><table cellspacing="0"><tr><td class="check"><input type="checkbox" /></td><td class="table"><a>test_1</a></td><td></td></tr><tr><td class="check"><input type="checkbox" /></td><td class="table"><a>test_2</a></td><td></td></tr></table></td></tr>');
 24 		}
 25 
 26 		// Expand/collapse row
 27 		if ($target.hasClass('expand')) {
 28 			that.collapse($row);
 29 		} else {
 30 			that.expand($row);
 31 		}
 32 
 33 		$('.context:visible').hide();
 34 
 35 		return false;
 36 	});
 37 
 38 	this.$body
 39 		.bind('all', function() { that.$body.find('td + td a').removeClass('partial'); })
 40 		.bind('nothing', function() { that.$body.find('td + td a').removeClass('partial'); });
 41 
 42 }
 43 
 44 /**
 45  * Expand row
 46  * 
 47  */
 48 Juxta.TreeGrid.prototype.expand = function(row) {
 49 	var $row = $(row);
 50 
 51 	$row.find('td.collapse').removeClass('collapse').addClass('expand');
 52 	$row.next('.content').hide();
 53 }
 54 
 55 /**
 56  * Collapse row
 57  * 
 58  */
 59 Juxta.TreeGrid.prototype.collapse = function(row) {
 60 	var $row = $(row);
 61 
 62 	$row.find('td.expand').removeClass('expand').addClass('collapse');
 63 	$row.next('.content').show();
 64 	if ($row.find('[type=checkbox]').is(':checked')) {
 65 		$row.next('tr.content').find('[type=checkbox]')
 66 			.attr('checked', true)
 67 			.parents('tr').find('a').addClass('checked');
 68 	}
 69 }
 70 
 71 /**
 72  * Select row
 73  * 
 74  */
 75 Juxta.TreeGrid.prototype.selectRow = function(row) {
 76 	this._selectRow(row);
 77 	var $row = $(row);
 78 
 79 	$row.find('a').removeClass('partial');
 80 
 81 	var childs = $row.parents('tr.content').find('[type=checkbox]').length,
 82 		childsChecked = $row.parents('tr.content').find('[type=checkbox]:checked').length;
 83 
 84 	// Check parent row if its child selected all
 85 	if (childs > 0) {
 86 		if (childs === childsChecked) {
 87 			$row.parents('tr.content').prev('tr')
 88 				.find('[type=checkbox]').attr('checked', true)
 89 				.parents('tr').find('a').removeClass('partial').addClass('checked');
 90 		} else if (childsChecked > 0) {
 91 			$row.parents('tr.content').prev('tr')
 92 				.find('a').addClass('partial');
 93 		}
 94 	}
 95 
 96 	// Check child rows
 97 	$row.next('tr.content')
 98 		.find('[type=checkbox]').attr('checked', true)
 99 		.parents('tr').find('a').addClass('checked');
100 }
101 
102 /**
103  * Deselect row
104  * 
105  */
106 Juxta.TreeGrid.prototype.deselectRow = function(row) {
107 	this._deselectRow(row);
108 	var $row = $(row);
109 
110 	var childs = $row.parents('tr.content').find('[type=checkbox]').length,
111 		childsChecked = $row.parents('tr.content').find('[type=checkbox]:checked').length;
112 
113 	// Uncheck parent row if its child slected none
114 	if (childs > 0) {
115 		if (childsChecked === 0) {
116 			$row.parents('tr.content').prev('tr')
117 				.find('a').removeClass('partial');
118 		} else if (childs > childsChecked) {
119 			$row.parents('tr.content').prev('tr')
120 				.find('[type=checkbox]').attr('checked', false)
121 				.parents('tr').find('a').removeClass('checked').addClass('partial');
122 		}
123 	}
124 
125 	// Uncheck child rows
126 	$row.next('tr.content')
127 		.find('[type=checkbox]').attr('checked', false)
128 		.parents('tr').find('a').removeClass('checked');
129 }
