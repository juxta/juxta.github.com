 32 			context: this
 33 		});

 40 		Juxta.request({
 41 			action: {get: 'collations'},
 42 			cache: true,
 43 			context: this,
 44 			success: this.responseGetCollations
 45 		});
 46 	},

 48 		var that = this;
 49 		if (!$.isEmptyObject(response.data)) {
 50 			var $collations = this.$collations;
 51 			this.$collations.empty().append('<option>Default</option>');
 52 			$.each(response.data, function(charset, collations) {
 53 				var $charset = $('<optgroup>').attr('label', charset);
 54 				if (!$.isEmptyObject(collations)) {
 55 					$.each(collations, function(index, charset) {
 56 						$charset.append($('<option>').text(charset));
 57 					});
 58 				}
 59 				that.$collations.append($charset);
 60 			});
 61 		}
 62 	}
 63 });
