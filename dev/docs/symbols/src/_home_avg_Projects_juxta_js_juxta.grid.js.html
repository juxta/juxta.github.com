1 /**
  2  * @class Grid
  3  */
  4 Juxta.Grid = function(grid) {
  5 	this.init(grid);
  6 }
  7 
  8 Juxta.Grid.prototype = {
  9 	statistics: {
 10 		item: 'item',
 11 		items: 'items',
 12 		all: 0,
 13 		selected: 0
 14 	},
 15 	content: null,
 16 	from: null,
 17 	cache: {},
 18 	init: function(grid) {
 19 		this.$container = $(grid);
 20 		this.$bodyContainer = this.$container.find('.body');
 21 		this.$body = this.$bodyContainer.find('table');
 22 		this.$notFound = this.$bodyContainer.find('.not-found')
 23 		this.head = this.$container.find('.head');
 24 		this.$actions = this.$container.find('.actions');
 25 		this.$context = this.$container.find('.context');
 26 
 27 		var that = this;
 28 
 29 		this.$body.change(function(event) {
 30 			// Select/deselect row
 31 			if ($(event.target).is('[type=checkbox]')) {
 32 				$('.context:visible').hide();
 33 
 34 				var $row = $(event.target).parent().parent();
 35 				if ($(event.target).is('[type=checkbox]:checked')) {
 36 					that.select($row);
 37 				} else{
 38 					that.deselect($row);
 39 				}
 40 			}
 41 
 42 			// Count all and selected items
 43 			that.statistics.all = that.$body.find('tr:not(tr tr):not(.content)').find('[type=checkbox]').length;
 44 			that.statistics.selected = that.$body.find('tr:not(tr tr):not(.content)').find('[type=checkbox]:checked').length;
 45 
 46 			// Change all, nothing links states
 47 			if (that.statistics.all > 0 && that.statistics.all == that.statistics.selected) {
 48 				that.$actions.find('.all').addClass('active');
 49 				that.$actions.find('.nothing').removeClass('active');
 50 			} else if(that.statistics.selected == 0) {
 51 				that.$actions.find('.all').removeClass('active');
 52 				that.$actions.find('.nothing').addClass('active');
 53 			} else{
 54 				that.$actions.find('.all').removeClass('active');
 55 				that.$actions.find('.nothing').removeClass('active');
 56 			}
 57 
 58 			// Disable group actions' buttons if nothing selected, enable else
 59 			if (that.statistics.selected < 1) {
 60 				that.$actions.find('input[type=button]').attr('disabled', true);
 61 			} else{
 62 				that.$actions.find('input[type=button]').attr('disabled', false);
 63 			}
 64 		});
 65 
 66 		// Trigger event with type equals action name
 67 		this.$actions.bind('click', function(event) {
 68 			var $button = $(event.target);
 69 			if ($button.is('span.like-a, input') && $button.attr('name')) {
 70 				that.$actions.trigger($button.attr('name'));
 71 				that.$body.trigger($button.attr('name'));
 72 			}
 73 		});
 74 
 75 		this.$actions.bind('all', function() {
 76 			that.select();
 77 		});
 78 		this.$actions.bind('nothing', function() {
 79 			that.deselect();
 80 		});
 81 
 82 		if (this.$context.is('.context')) {
 83 			this.contextMenu = new Juxta.ContextMenu(this.$body, this.$container.find('.context'));
 84 		}
 85 
 86 	},
 87 	height: function(height) {
 88 		if (height) {
 89 			this.$notFound.css('top', height / 2 - 14 + 'px');
 90 		}
 91 		return this.$bodyContainer.height(height);
 92 	},
 93 	prepare: function(template) {
 94 		if (template) {
 95 			var that = this;
 96 
 97 			this.$actions.empty();
 98 			if (template['actions']) {
 99 				this.$actions.html(template['actions']);
100 			}
101 
102 			// Empty grid header and body
103 			if (template['head']) {
104 				this.empty();
105 				this.head.empty().show();
106 				this.$bodyContainer.show();
107 				this.$container.find('.proper').hide();
108 			} else {
109 				this.empty();
110 				this.head.empty().hide();
111 				this.$bodyContainer.hide();
112 			}
113 
114 			// Make grid header
115 			if (template['head']) {
116 				this.head.empty();
117 				$.each(template['head'], function(i, value) {
118 					that.head.append('<li class="' + i + '">' + value + '</li>');
119 				});
120 			}
121 
122 			// Define context for status bar
123 			if ($.isArray(template.context[0])) {
124 				this.statistics.item = template.context[0][0];
125 				this.statistics.items = template.context[0][1];
126 			} else{
127 				this.statistics.item = template.context[0];
128 				this.statistics.items = 'items';
129 			}
130 
131 			//
132 			this.statistics.all = 0;
133 			this.$body.trigger('change');
134 
135 			return true;
136 		} else {
137 			return false;
138 		}
139 	},
140 	fill: function(data) {
141 		var that = this;
142 
143 		this.empty();
144 		this.content = data.contents;
145 		if (data.from) {
146 			this.from = data.from;
147 		}
148 		if (data && data.data && (data.data.length > 0 || $.isPlainObject(data.data))) {
149 			this.statistics.all = data.data.length;
150 
151 			var template = data.row;
152 			jQuery.each(data.data, function(i, value) {
153 				if ($.isPlainObject(data.data)) {
154 					value = [i, value];
155 				}
156 				var forTemplate = {},
157 					cacheName;
158 				jQuery.each(data.context, function(j, valueName) {
159 					var name;
160 					if (data.context.length == 1) {
161 						if ($.isArray(valueName)) {
162 							name = valueName[0];
163 						} else{
164 							name = valueName;
165 						}
166 						forTemplate[name] = value;
167 					} else{
168 						if ($.isArray(valueName)) {
169 							name = valueName[0];
170 						} else{
171 							name = valueName;
172 						}
173 						forTemplate[name] = value[j];
174 					}
175 					if (!cacheName) {
176 						cacheName = name;
177 					}
178 				});
179 				$.extend(forTemplate, {database: data['from']});
180 
181 				var $q = $($.template(template, forTemplate)).appendTo(that.$body);
182 				that.cache[forTemplate[cacheName]] = $q;
183 			});
184 			this.$body.trigger('change');
185 
186 			// Make context menu
187 			if (data.contextMenu) {
188 				this.$container.find('.context ul').html($.template(data.contextMenu, {database: data['from']}));
189 			}
190 		} else {
191 			this.$notFound.css('top', this.$container.find('.body').height() / 2 - 14 + 'px').show();
192 		}
193 	},
194 	/**
195 	 * Empty grid body
196 	 */
197 	empty: function() {
198 		this.$body.empty();
199 		this.$notFound.hide();
200 		this.cache = {};
201 		this.content = null;
202 		this.from = null;
203 		this.statistics.all = 0;
204 		this.statistics.selected = 0;
205 	},
206 	/**
207 	 * Select rows
208 	 */
209 	select: function(row) {
210 		if (row) {
211 			this.selectRow(row);
212 		} else {
213 			this.selectAll();
214 		}
215 	},
216 	/**
217 	 * Deselect rows
218 	 */
219 	deselect: function(row) {
220 		if (row) {
221 			this.deselectRow(row);
222 		} else {
223 			this.deselectAll();
224 		}
225 	},
226 	/**
227 	 * Select all rows
228 	 */
229 	selectAll: function() {
230 		$('.context:visible').hide();
231 		this.$body.find('input[type=checkbox]').attr('checked', 'checked').parent().next('td').find('a').addClass('checked');
232 		this.$body.trigger('change');
233 	},
234 	/**
235 	 * Deselect all rows
236 	 */
237 	deselectAll: function() {
238 		this.$body.find('input[type=checkbox]').removeAttr('checked', '');
239 		this.$body.find('a.checked').removeClass('checked');
240 		this.$body.trigger('change');
241 	},
242 	/**
243 	 * Select one row
244 	 */
245 	selectRow: function(row) {
246 		var $row = $(row);
247 
248 		// Highlight link
249 		$row.find('td.check').next('td').find('a').addClass('checked');
250 	},
251 	/**
252 	 * Deselect one row
253 	 */
254 	deselectRow: function(row) {
255 		var $row = $(row);
256 
257 		// Unhighlight link
258 		$row.find('td.check').next('td').find('a').removeClass('checked');
259 	},
260 	/**
261 	 * Returns names of selected rows
262 	 */
263 	selected: function(filter, group) {
264 		var selected = this.$body.find('input[type=checkbox]:checked');
265 
266 		// Apply filter
267 		if (filter) {
268 			selected = selected.filter(filter);
269 		}
270 
271 		// Collect to arrya and group by attribute
272 		if (group) {
273 			var grouped = {};
274 			jQuery.each(selected, function() {
275 				if (grouped[$(this).attr(group)] === undefined) {
276 					grouped[$(this).attr(group)] = [];
277 				}
278 				grouped[$(this).attr(group)].push($(this).attr('name'));
279 			});
280 			selected = grouped;
281 		} else {
282 			selected = selected.map(function() { return $(this).attr('name'); }).toArray();
283 		}
284 
285 		return !$.isEmptyObject(selected) ? selected : null;
286 	},
287 	/**
288 	 * Removes rows by name
289 	 */
290 	remove: function(names, filter) {
291 		var that = this;
292 
293 		if (!$.isArray(names)) {
294 			names = [names];
295 		}
296 
297 		$.each(names, function(i, name) {
298 			if (that.cache[name] && (filter === undefined || that.cache[name].find('.check :checkbox').is(filter))) {
299 				that.cache[name].remove();
300 			}
301 		});
302 
303 		that.$body.trigger('change');
304 	}
305 }
