1 Juxta.FloatBox = function(element, options) {
  2 	this.init(element, options);
  3 }
  4 
  5 Juxta.FloatBox.prototype = {
  6 	settings: {
  7 		title: 'New window',
  8 		closable: true,
  9 		center: true
 10 	},
 11 	init: function(element, options) {
 12 		var that = this;
 13 
 14 		this.settings = $.extend({}, this.settings, options);
 15 
 16 		this.$container = $(element);
 17 
 18 		// @depricated
 19 		this.$floatBox = this.$container;
 20 
 21 		// Head
 22 		if (this.$container.find('h3').is('h3')) {
 23 			this.$head = this.$container.find('h3');
 24 		} else {
 25 			this.$head = this.$container.prepend('<h3>'+ this.settings.title + '</h3>').find('h3');
 26 		}
 27 
 28 		this.settings.title = this.$head.html();
 29 
 30 		// Close button
 31 		if (this.$container.find('input[type=button].close').is('input')) {
 32 			this.$terminator = this.$container.find('input[type=button].close');
 33 		} else {
 34 			this.$terminator = $('<input type="button" class="close"/>').insertAfter(this.$head).attr('disabled', !this.settings.closable);
 35 		}
 36 
 37 		this.$terminator.click(function() { that.hide(); });
 38 
 39 		// Drag options
 40 		this.$container.draggable({scroll: false, handle: 'h3'});
 41 
 42 		this.center();
 43 	},
 44 	show: function(options, content) {
 45 		options = $.extend({}, this.settings, options);
 46 
 47 		if (options.name) {
 48 			options.name = '<a href="">' + options.name + '</a>'; 
 49 		}
 50 		if (options.from) {
 51 			options.from = '<span class="from">from <a>' + options.from + '</a></span>'; 
 52 		}
 53 
 54 		this.$head.html($.template(options.title, options));
 55 
 56 		// Append content
 57 		if (content) {
 58 			this.clear();
 59 			$(content).insertAfter(this.$terminator);
 60 		}
 61 
 62 		this.$container.show();
 63 
 64 		if (options.center) {
 65 			this.center();
 66 		}
 67 	},
 68 	hide: function() {
 69 		this.$floatBox.hide();
 70 	},
 71 	/**
 72 	 * Center box
 73 	 */
 74 	center: function() {
 75 		var body = {
 76 			height: $(document.body).height(),
 77 			width: $(document.body).width()
 78 		};
 79 
 80 		this.$floatBox.css({
 81 			left: (body.width - this.$floatBox.width()) / 2,
 82 			top: parseInt(0.75 * (body.height - this.$floatBox.height()) / 2)
 83 		});
 84 	},
 85 	/**
 86 	 * Clear content
 87 	 */
 88 	clear: function() {
 89 		this.$container.find('> *:not(h3):not([type=button].close)').remove();
 90 	}
 91 };
