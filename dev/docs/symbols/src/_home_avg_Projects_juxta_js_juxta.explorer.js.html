  8 		this._super(element);

 18 		this.grid.$actions.bind('drop', function() {
 19 			var params = {
 20 					drop: that.grid.content,
 21 					item: that.grid.statistics.item,
 22 					items: that.grid.statistics.items
 23 				};
 24 
 25 			// Stored routines group by attribute 'routine'
 26 			if (that.grid.content === 'routines') {
 27 				params[params.drop] = that.grid.selected(null, 'routine');
 28 			} else {
 29 				params[params.drop] = that.grid.selected();
 30 			}
 31 
 32 			// Set from
 33 			if (that.grid.from) {
 34 				params['from'] = that.grid.from;
 35 			}
 36 
 37 			that.drop(params);
 38 		});
 39 
 40 		this.grid.$actions.bind('kill', function() {
 41 			that.kill({processes: that.grid.selected()});
 42 		});
 43 
 44 		this.grid.$body.bind('change', function() {
 45 			var status = '';
 46 			if (that.grid.statistics.all > 0) {
 47 				status += that.grid.statistics.all;
 48 				if (that.grid.statistics.all == 1) {
 49 					status += ' ' + that.grid.statistics.item;
 50 				} else {
 51 					status += ' ' + that.grid.statistics.items;
 52 				}
 53 			}
 54 			Juxta.explorer.status(status);
 55 		});

 57 	show: function(options) {

 62 		var _this = event && event.data._this || this;

 67 	request: function(params) {
 68 		if (this.templates[params.show]['head']['header']['from'] === null) {
 69 			this.templates[params.show]['head']['header']['from'] = params.from;
 70 		}
 71 		this.show(this.templates[params.show]['head']);
 72 		// Extend request options
 73 		if (this.templates[params.show].query) {
 74 			params = $.extend({}, this.templates[params.show].query, params);
 75 		}
 76 		// Move options values from query to options variable
 77 		var query = $.extend({}, params),
 78 			options = {};
 79 
 80 		$.each(['cache', 'index', 'refresh'], function(index, value) {
 81 			delete query[value];
 82 			if (params[value] !== undefined) {
 83 				options[value] = params[value];
 84 			}
 85 		});
 86 		//
 87 		this.cache = Juxta.queryString(query);
 88 		if (this.prepare(query.show)) {
 89 			Juxta.request($.extend(
 90 				{},
 91 				{action: query, context: this, success: this.response},
 92 				this.settings,
 93 				options
 94 			));
 95 		}
 96 	},
 97 	response: function(data) {
 98 		this.show();
 99 		if (this.preparedFor == data.contents) {
100 			$.extend(data, this.templates[data.contents].grid);
101 			this.grid.fill(data);
102 		}
103 	},
104 	prepare: function(template) {
105 		if (template === this.preparedFor) {
106 			return true;
107 		} else if (this.grid.prepare(this.templates[template].grid)) {
108 			this.preparedFor = template;
109 			return true;
110 		} else {
111 			return false;
112 		}
113 	},
114 	drop: function(params) {
115 		var message = '',
116 			countItemsForDrop = 0,
117 			singleItemForDrop = null;
118 
119 		// Count items for drop
120 		if (typeof params[params.drop] === 'string') {
121 			countItemsForDrop = 1;
122 			singleItemForDrop = params[params.drop];
123 		} else if ($.isArray(params[params.drop])) {
124 			countItemsForDrop = params[params.drop].length;
125 			if (countItemsForDrop === 1) {
126 				singleItemForDrop = params[params.drop][0];
127 			}
128 		} else if ($.isPlainObject(params[params.drop])) {
129 			$.each(params[params.drop], function() {
130 				if ($.isArray(this)) {
131 					countItemsForDrop += this.length;
132 					if (countItemsForDrop === 1) {
133 						singleItemForDrop = this[0];
134 					}
135 				}
136 			});
137 		}
138 
139 		// Compose message
140 		if (countItemsForDrop === 1) {
141 			message = 'Drop ' + params.item + ' `' + singleItemForDrop + '`';
142 		} else {
143 			message = 'Drop ' + countItemsForDrop + ' ' + params.items;
144 		}
145 
146 		//
147 		if (Juxta.confirm(message + '?')) {
148 			this.requestDrop(params);
149 		}
150 	},
151 	requestDrop: function(params) {
152 		var action = {drop: params.drop},
153 			data = {};
154 
155 		//
156 		if (params.from) {
157 			action['from'] = params.from
158 		}
159 		data[params.drop] = params[params.drop];
160 
161 		Juxta.request({
162 			action: action,
163 			data: data,
164 			success: this.responseDrop,
165 			error: this.responseDrop,
166 			context: this
167 		});
168 	},
169 	responseDrop: function(response) {
170 		this.grid.deselect();
171 
172 		// @todo Remove `response.dropped.procedure` etc.
173 		if ($.isPlainObject(response.dropped)) {
174 			if (response.dropped['procedure']) {
175 				this.grid.remove(response.dropped['procedure'], '[routine=procedure]');
176 			}
177 			if (response.dropped['function']) {
178 				this.grid.remove(response.dropped['function'], '[routine=function]');
179 			}
180 		} else {
181 			this.grid.remove(response.dropped);
182 		}
183 
184 		Juxta.cache.flush(this.cache);
185 	},
