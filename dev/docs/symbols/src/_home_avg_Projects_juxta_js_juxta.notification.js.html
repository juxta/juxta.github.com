1 Juxta.Notification = $.Class();

 23 			this.container.empty();
 24 		}
 25 		if (options.element) {
 26 			var notify = options.element;
 27 		} else {
 28 			var notify = $('<li><span></span></li>').appendTo(this.container);
 29 		}
 30 		notify.show().find('span').text(message);
 31 		if (options.hide) {
 32 			this.hide(notify, options);
 33 		}

 35 		return notify;
 36 	},

 38 		if (arguments.length == 2) {
 39 			element.delay(options.delay).slideUp(options.hideSpeed, function() { $(this).remove(); });
 40 			this.load = null;
 41 		} else {
 42 			this.load = null;
 43 			this.loads = 0;
 44 			this.container.empty();
 45 		}
 46 	},
 47 	loading: function(message, options) {
 48 		var self = this;
 49 		options = $.extend({}, self.loadingSettings, options);
 50 		if (message === false && this.load) {
 51 			if (--this.loads == 0) {
 52 				this.hide(this.load, options);
 53 			}
 54 		} else if(message !== false) {
 55 			if (this.loads++ == 0) {
 56 				this.container.empty();
 57 				message = message || 'Loading..'; 
 58 				if (this.load) {
 59 					options.element = this.load;
 60 				}
 61 				this.load = this.show(message, options);
 62 			}
 63 		}
 64 	}
 65 };
