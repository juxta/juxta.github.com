  3 	tree: {},

  9 			'connection': this.$sidebar.find('li.connection span.value'),

 11 			'table': this.$sidebar.find('li.table span.value'),
 12 		}

 15 			$(this).html('<span>' + $(this).html() + '</span>')

 17 					$(this).addClass($(this).attr('href').replace(/#/g, ''))

 24 			if ($head.is(':not(.fold):not(.last):visible')) {
 25 				$head.addClass('fold').find('.buttons').slideUp(250);
 26 				self.$sidebar.find('.last .buttons').slideDown('250');
 27 			} else if ($head.is('.fold')) {
 28 				self.$sidebar.find('ul:first-child > li:not(.fold):not(.last):visible').addClass('fold')
 29 					.find('.buttons').slideUp(250);
 30 				self.$sidebar.
 31 					find('ul:first-child > li.last').
 32 					find('.buttons').slideUp(250);
 33 				$head.removeClass('fold').find('.buttons').slideDown(250);
 34 			} else if ($head.is('.last') && $head.find('.buttons').not(':visible')) {
 35 				$head.find('.buttons').slideDown(250);
 36 				self.$sidebar.find('ul:first-child > li:not(.fold):not(.last):visible').addClass('fold')
 37 					.find('.buttons').slideUp(250);
 38 			}
 39 		});

 41 		this.$sidebar.hover(
 42 			function() {
 43 				clearTimeout(this.timer);
 44 			},

 47 			}
 48 		);
 49 	},

 52 		this.$sidebar
 53 			.find('ul:first-child > li:not(.fold):not(.last):visible').addClass('fold').find('.buttons').slideUp(250);
 54 		this.$sidebar
 55 			.find('.last .buttons').slideDown(250);
 56 	},
 57 	highlight: function(link, path) {

 59

 61 			this.path(path);
 62 		}
 63

 65 			var level = this.$sidebar.find('ul:first-child > li.' + this.tree[link]);
 66 			if (level.is('.connection')) {
 67 				this.heads.filter('.connection').addClass('last').show().removeClass('fold').find('.buttons').show();
 68 				this.heads.not('.connection').removeClass('last').hide();
 69 			} else if (level.is('.database')) {
 70 				this.heads.filter('.connection').removeClass('last').show().addClass('fold').find('.buttons').hide();
 71 				this.heads.filter('.database').addClass('last').show().removeClass('fold').find('.buttons').show();
 72 				this.heads.filter('.table').removeClass('last').hide();
 73 			} else if (level.is('.table')) {
 74 				this.heads.filter('.connection').removeClass('last').show().addClass('fold').find('.buttons').hide();
 75 				this.heads.filter('.database').removeClass('last').show().addClass('fold').find('.buttons').hide();
 76 				this.heads.filter('.table').addClass('last').show().removeClass('fold').find('.buttons').show();
 77 			}
 78 		}
 79 		this.$sidebar.find('.buttons li').removeClass('active');
 80 		this.$sidebar.find('li.' + link).addClass('active');
 81 	},
 82 	path: function(path) {
 83 		var self = this;
 84 		$.extend(self.path, path);
 85 		$.each(self.values, function(item) {
 86 			$(this).text(self.path[item]);
 87 		});
 88 		this.repairLinks();
 89 	},
 90 	repairLinks: function() {
 91 		var self = this;
 92 		$('#sidebar').find('li.database a').each(function() {
 93 			this.href = '#' + self.path.database + '/' + this.className;
 94 		});
 95 		$('#sidebar').find('li.table a').each(function() {
 96 			this.href = '#' + self.path.database + '/' + self.path.table + '/' + this.className;
 97 		});
 98 	}
 99 };
