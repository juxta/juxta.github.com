1 Juxta.Auth = $.Class(Juxta.FloatBox, {
  2 	storedConnections: undefined,
  3 	init: function(element) {
  4 		this._super(element, {title: 'Connect to MySQL Server', closable: false});
  5 		this.$form = this.$floatBox.find('form[name=login]');
  6 		this.form = this.$form.get(0);
  7 		this.$connections = $('select[name=connection]');
  8 		this.$password = this.$form.find('input[type=password]');
  9 		this.$submit = this.$form.find('input[type=submit]');
 10 
 11 		var self = this;
 12 		$('#header a[href=#logout]').click(function() {
 13 			self.logout();
 14 			return false;
 15 		});
 16 		this.$form.bind('submit', function() {
 17 			self.$submit.focus();
 18 			self.login();
 19 			return false;
 20 		});
 21 		this.$connections.bind('change', function() {
 22 			self.fillForm(self.storedConnections[this.value]);
 23 			self.$connections.find('option[value=0]').remove();
 24 			self.$password.focus().val(null);
 25 		});
 26 		$('input[name=host],input[name=user]', this.$form).bind('keyup', function() {
 27 			if (self.$connections.val() > 0) {
 28 				var curConnection = self.storedConnections[self.$connections.val()];
 29 				if (curConnection['host'] != self.form['host'].value ||
 30 					curConnection['user'] != self.form['user'].value
 31 				) {
 32 					self.$connections.prepend('<option value="0"></option>').val(0);
 33 				}
 34 			}
 35 		});
 36 	},
 37 	show: function() {
 38 		if (!this.storedConnections) {
 39 			Juxta.request({action: {get: 'connections'}, context: this, success: this.getConnectionsResponse});
 40 		}
 41 		//
 42 		Juxta.hide();
 43 		this.$submit.attr('disabled', false);
 44 		this.$password.val(null);
 45 		this._show();
 46 		if (this.$form[0]['host'].value && this.$form[0]['user'].value) {
 47 			this.$password.focus();
 48 		}
 49 	},
 50 	login : function() {
 51 		if (jQuery.trim($('input[name=host]', this.$form).val()) == '') {
 52 			$('input[name=host]', this.$form).val('localhost');
 53 		}
 54 		if (jQuery.trim($('input[name=port]', this.$form).val()) == '') {
 55 			$('input[name=port]', this.$form).val('3306');
 56 		}
 57 		//
 58 		this.$submit.attr('disabled', true);
 59 		Juxta.request({
 60 			action: 'login',
 61 			data: this.$form.serialize(),
 62 			loading: 'Connecting to ' + $('input[name=host]', this.$form).val(),
 63 			success: this.loginResponse,
 64 			context: this
 65 		});
 66 	},
 67 	loginResponse: function(response) {
 68 		if (response.result == 'connected') {
 69 			Juxta.state = null;
 70 			document.location.hash = '#databases';
 71 		} else {
 72 			Juxta.error(response.message);
 73 			//
 74 			this.$submit.attr('disabled', false);
 75 			this.$password.focus();
 76 		}
 77 	},
 78 	logout: function() {
 79 		Juxta.request({action: 'logout', success: function() { Juxta.cache.flush(); document.location.hash = '#login'; }});
 80 	},
 81 	getConnectionsResponse: function(response) {
 82 		if (!$.isEmptyObject(response.data)) {
 83 			var self = this, connections = {};
 84 			$.each(response.data, function(i) {
 85 				if (this.name == undefined) {
 86 					this.name = this.user + '@' + this.host;
 87 				}
 88 				$('<option>', {val: i + 1, selected: this['default']}).html(this.name).appendTo(self.$connections);
 89 				connections[i + 1] = this;
 90 			});
 91 			this.$connections.attr('disabled', false);
 92 			//
 93 			this.storedConnections = connections;
 94 			if (this.form['host'].value == '' && this.form['user'].value == '') {
 95 				this.fillForm(this.storedConnections[this.$connections.val()]);
 96 				this.$password.focus();
 97 			} else {
 98 				this.$connections.prepend('<option value="0"></option>');
 99 			}
100 		}
101 	},
102 	fillForm: function(values) {
103 		if (values) {
104 			this.form['host'].value = values['host'];
105 			this.form['port'].value = values['port'];
106 			this.form['user'].value = values['user'];
107 		}
108 	}
109 });
