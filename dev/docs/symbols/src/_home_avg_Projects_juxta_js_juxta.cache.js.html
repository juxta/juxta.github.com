1 /**
  2  * @class Cache
  3  */
  4 Juxta.Cache = function() {
  5 
  6 	var defaultLifeTime = 10000;
  7 
  8 	this.cache = {};
  9 
 10 	this.get = function(key) {
 11 		var timestamp = (new Date()).getTime();
 12 
 13 		if (this.cache[key] && this.cache[key]['expire'] >= timestamp) {
 14 			return this.cache[key]['data'];
 15 		} else if (this.cache[key]) {
 16 			this.flush(key);
 17 			return undefined;
 18 		} else {
 19 			return undefined;
 20 		}
 21 	}
 22 
 23 	this.set = function(key, data, lifeTime, index) {
 24 		var expire,
 25 			timestamp = (new Date()).getTime();
 26 		//
 27 		if (lifeTime === 0 || lifeTime === Infinity) {
 28 			expire = Infinity;
 29 		} else if (lifeTime === true || lifeTime === undefined) {
 30 			expire = timestamp + defaultLifeTime;
 31 		} else if (Number(lifeTime) > 0) {
 32 			expire = timestamp + lifeTime * 1000;
 33 		}
 34 
 35 		if (expire) {
 36 			this.cache[key] = {data: data, expire: expire};
 37 			if (index) {
 38 				this.index(key, index);
 39 			}
 40 			return true;
 41 		} else {
 42 			return false;
 43 		}
 44 	}
 45 
 46 	this.flush = function(key) {
 47 		if (key) {
 48 			delete this.cache[key];
 49 		} else {
 50 			this.cache = {};
 51 		}
 52 	}
 53 
 54 	/**
 55 	 * Index data in cache
 56 	 * @param {String} key
 57 	 * @param {Object} params
 58 	 */
 59 	this.index = function(key, params) {
 60 		if (this.cache[key]) {
 61 			var data = this.cache[key]['data'],
 62 				path = params.path;
 63 			if (path && typeof path !== 'object') {
 64 				data = data[path];
 65 			} else if (path) {
 66 				for (var i in path) {
 67 					if (data[path[i]]) {
 68 						data = data[path[i]];
 69 					}
 70 				}
 71 			}
 72 
 73 			if (this.cache[key]['index'] === undefined) {
 74 				this.cache[key]['index'] = {};
 75 			}
 76 
 77 			if (params.name && params.field !== undefined) {
 78 				var field = params.field,
 79 					index = params.name;
 80 				this.cache[key]['index'][index] = {};
 81 				for (var row in data) {
 82 					if (data[row][field] !== undefined) {
 83 						if (path !== undefined) {
 84 							this.cache[key]['index'][index][data[row][field]] = [row, path];
 85 						} else {
 86 							this.cache[key]['index'][index][data[row][field]] = row;
 87 						}
 88 					}
 89 				}
 90 			}
 91 		}
 92 	}
 93 
 94 	/**
 95 	 * Search data using index
 96 	 * @params {String} key
 97 	 * @params {String} search
 98 	 */
 99 	this.search = function(key, search) {
100 		if (this.cache[key]['index']) {
101 			if (typeof search === 'object') {
102 				for (var index in search) break;
103 				if (index && this.cache[key]['index'][index] && this.cache[key]['index'][index][search[index]]) {
104 					if (typeof this.cache[key]['index'][index][search[index]] === 'object') {
105 						var path = this.cache[key]['index'][index][search[index]][1],
106 							row = this.cache[key]['index'][index][search[index]][0];
107 					} else {
108 						var row = this.cache[key]['index'][index][search[index]][0];
109 					}
110 					var data = this.cache[key]['data'];
111 					if (path) {
112 						for (var i in path) {
113 							if (data[path[i]]) {
114 								data = data[path[i]];
115 							}
116 						}
117 					}
118 					if (row && data[row]) {
119 						return data[row];
120 					}
121 				}
122 			}
123 		}
124 	}
125 
126 }
